# ============================================
# INSTALLATION INSTRUCTIONS
# ============================================
#
# 1. Install JDK 23:
#    - Windows: Download from Oracle or use Chocolatey: choco install openjdk23
#    - Linux: sudo apt-get install openjdk-23-jdk (Ubuntu/Debian)
#             sudo yum install java-23-openjdk-devel (RHEL/CentOS)
#    - macOS: brew install openjdk@23
#
# 2. Verify Java installation:
#    java -version
#    javac -version
#
# 3. Install Maven (if not using Maven Wrapper):
#    - Windows: Download from Apache or use Chocolatey: choco install maven
#    - Linux: sudo apt-get install maven
#    - macOS: brew install maven
#
# 4. Verify Maven installation:
#    mvn -version
#
# 5. Install PostgreSQL:
#    - Windows: Download installer from PostgreSQL website
#    - Linux: sudo apt-get install postgresql postgresql-contrib
#    - macOS: brew install postgresql
#
# 6. Start PostgreSQL service:
#    - Windows: Use Services panel or: net start postgresql-x64-XX
#    - Linux: sudo systemctl start postgresql
#    - macOS: brew services start postgresql
#
# 7. Create database:
#    psql -U postgres
#    CREATE DATABASE internship_portal;
#    \q
#
# 8. Run database schema:
#    psql -U postgres -d internship_portal -f Database/schema.sql
#

# ============================================
# TROUBLESHOOTING
# ============================================
#
# Issue: "Java version not found"
# Solution: Verify JAVA_HOME environment variable is set correctly
#
# Issue: "Maven command not found"
# Solution: Use Maven Wrapper (./mvnw or mvnw.cmd) or add Maven to PATH
#
# Issue: "Cannot connect to database"
# Solution: Verify PostgreSQL is running and credentials in application.properties are correct
#
# Issue: "Port 8080 already in use"
# Solution: Change server.port in application.properties or stop the conflicting service
